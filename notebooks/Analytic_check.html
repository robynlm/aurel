err_max = 1e-4

custom_funcs = {
    "hyper": lambda a, b, z: sc.hyp2f1(a[0], a[1], b[0], z),    "exp_polar": np.exp
}

def compare_values(theory, numerical):
    """Compare analytical and numerical values, returning the maximum error.

    Args:
        theory: Analytical expression (SymPy).
        numerical: Numerical array (NumPy).

    Returns
    -------
        error: Maximum relative error (float).
    """
    f_num = sp.lambdify((t, x, y, z), theory, modules=[custom_funcs, "numpy"])
    s1 = np.real(f_num(nt, fd.cutoffmask(nx), fd.cutoffmask(ny), fd.cutoffmask(nz)))
    s2 = fd.cutoffmask(numerical)
    if theory != 0:
        rel_err = np.abs(aurel.maths.safe_division(s2 - s1, s1))
        error = np.max(np.ma.masked_invalid(rel_err))
    else:
        error = np.max(np.abs(s2))
    return error

def checks(var, theory, numerical):
    """Print error information for a variable if above threshold.

    Args:
        var: Variable name (str).
        theory: Analytical value.
        numerical: Numerical value.
    """
    error = compare_values(theory, numerical)
    if error > err_max:
        print(var,
              ", max(abs(num_val)) = ", np.max(abs(fd.cutoffmask(numerical))),
              ", relative error =", error)

def check1d(dim, var, theory, numerical):
    """Check 1D arrays element-wise and print errors above threshold.

    Args:
        dim: Dimension size (int).
        var: Variable name (str).
        theory: Analytical array.
        numerical: Numerical array.
    """
    for i in range(dim):
        checks(var+f"{i}", theory[i], numerical[i])

def check2d(dim, var, theory, numerical):
    """Check 2D arrays element-wise and print errors above threshold.

    Args:
        dim: Dimension size (int).
        var: Variable name (str).
        theory: Analytical array.
        numerical: Numerical array.
    """
    for i in range(dim):
        for j in range(dim):
            checks(var+f"{i}{j}", theory[i, j], numerical[i, j])

def check3d(dim, var, theory, numerical):
    """Check 3D arrays element-wise and print errors above threshold.

    Args:
        dim: Dimension size (int).
        var: Variable name (str).
        theory: Analytical array.
        numerical: Numerical array.
    """
    for i in range(dim):
        for j in range(dim):
            for k in range(dim):
                checks(var+f"{i}{j}{k}", theory[i, j, k], numerical[i, j, k])

def check4d(dim, var, theory, numerical):
    """Check 4D arrays element-wise and print errors above threshold.

    Args:
        dim: Dimension size (int).
        var: Variable name (str).
        theory: Analytical array.
        numerical: Numerical array.
    """
    for i in range(dim):
        for j in range(dim):
            for k in range(dim):
                for h in range(dim):
                    checks(var+f"{i}{j}{k}{h}", theory[i, j, k, h], numerical[i, j, k, h])

def plot_them(analytical, numerical, cutoff=True):
    """Plot analytical vs numerical results and their relative error.

    Args:
        analytical: Analytical expression (SymPy).
        numerical: Numerical array (NumPy).
        cutoff: Whether to apply cutoff mask (bool, default True).
    """
    f_num = sp.lambdify((t, x, y, z), analytical, modules=[custom_funcs, "numpy"])
    A = np.real(f_num(nt, nx, ny, nz))
    if isinstance(A, int) or isinstance(A, float):
        A = np.full_like(numerical, A)
    rel_error = np.abs(aurel.maths.safe_division(numerical - A, A))
    ix, iy, iz = np.where(rel_error == np.max(rel_error))
    if len(ix) > 1:
        ix = ix[0]
        iy = iy[0]
        iz = iz[0]
    print(f"max {np.max(rel_error):.2e} error at", ix, iy, iz)
    if iy < 2*fd.mask_len or iz < 2*fd.mask_len:
        cutoff = False
        print('Max relative difference in boundary region')
    if cutoff:
        xv = fd.cutoffmask(fd.xarray)
        nv = fd.cutoffmask(numerical[:,iy,iz])
        av = fd.cutoffmask(A[:,iy,iz])
        rv = fd.cutoffmask(rel_error[:,iy,iz])
    else:
        xv = fd.xarray
        nv = numerical[:,iy,iz]
        av = A[:,iy,iz]
        rv = rel_error[:,iy,iz]
    plt.figure()
    plt.subplot(211)
    plt.plot(xv, nv,
             label="numerical")
    plt.plot(xv, av,
             linestyle='--',
             label="analytical")
    plt.grid()
    plt.legend()
    plt.subplot(212)
    plt.semilogy(xv, rv)
    plt.grid()
    plt.ylabel("relative difference")
    plt.xlabel("x")
    plt.subplots_adjust(hspace=0.0)
